## Goal

Implement two-stage event extraction pipeline that:
1. Takes logic schema as input (from LOGIC-01)
2. Extracts specific events from news text
3. Maps events to logic_ids
4. Outputs structured events with strength, validity, keywords

**In scope:**
- LLM prompt design for event extraction (Stage 2)
- Event schema with strength_raw, validity_period, keywords
- Event persistence to database
- Association to existing logic_ids
- Batch processing for multiple news articles

**Out of scope:**
- Fingerprint validation (LOGIC-04)
- Scoring rules application (LOGIC-03)

```python
class Event(Base):
    __tablename__ = "events"
    
    id = Column(Integer, primary_key=True)
    event_id = Column(String(64), unique=True, nullable=False, index=True)
    logic_id = Column(String(64), ForeignKey("logics.logic_id"), nullable=False, index=True)
    
    # Event metadata
    event_date = Column(Date, nullable=False)
    created_at = Column(DateTime, default=datetime.utcnow)
    source = Column(String(50), nullable=False)  # News source
    headline = Column(String(500), nullable=True)
    content_hash = Column(String(64), nullable=True)  # For deduplication
    
    # Scoring
    strength_raw = Column(Numeric(3, 2), nullable=False)  # 0.0-1.0
    direction = Column(Enum(LogicDirection), nullable=False)  # Inherited from logic
    
    # Validity
    validity_start = Column(Date, default=date.today)
    validity_end = Column(Date, nullable=True)  # Computed from logic.validity_days
    is_expired = Column(Boolean, default=False)
    
    # Relationships
    logic = relationship("Logic", back_populates="events")
    
    def __repr__(self):
        return f"<Event(event_id={self.event_id}, logic_id={self.logic_id}, strength={self.strength_raw})>"
```

### 2. Event Extraction Service (`src/logic/event_extractor.py`)

```python
class EventExtractionService:
    """Stage 2: Extract events from news using logic schema"""
    
    SYSTEM_PROMPT = """你是一名 A 股事件提取专家。

给定逻辑 schema 和新闻文本，提取具体事件：

逻辑 schema:
{logic_schema_json}

新闻文本：
{news_text}

输出 JSON 格式：

{{
  "events": [
    {{
      "event_id": "<auto-generated>",
      "logic_id": "<匹配的 logic_id>",
      "event_date": "<YYYY-MM-DD>",
      "headline": "<事件标题，20-50 字>",
      "strength_raw": <0.0-1.0>,
      "validity_days": <有效期天数，默认 30>,
      "keywords": ["<关键词 1>", "<关键词 2>"]
    }}
  ]
}}

强度评分标准：
- 0.8-1.0: 国家级政策、重大技术突破、业绩翻倍
- 0.5-0.7: 行业级新闻、重要财报、大额订单
- 0.3-0.4: 媒体报道、分析师点评、常规新闻
- 0.1-0.2: 传闻、不确定性高的信息

注意：
1. 只提取与给定逻辑 schema 相关的事件
2. 如果新闻不涉及任何给定逻辑，返回空数组
3. 事件日期从新闻中推断，格式 YYYY-MM-DD
4. strength_raw 必须基于客观标准，不要过度自信

输出（仅 JSON，无其他文字）："""

    async def extract_events(
        self, 
        news_text: str, 
        logics: List[Logic],
        source: str = None
    ) -> List[Event]:
        """Extract events from news, mapped to given logics"""
        
        if not logics:
            logger.debug("No logics provided, skipping event extraction")
            return []
        
        # Build logic schema JSON
        logic_schema = [
            {
                "logic_id": l.logic_id,
                "logic_family": l.logic_family,
                "direction": l.direction.value,
                "importance_level": l.importance_level.value,
                "description": l.description,
                "keywords": l.keywords,
                "validity_days": l.validity_days
            }
            for l in logics
        ]
        
        # Call LLM
        response = await self._call_llm(news_text, logic_schema)
        
        # Parse and validate
        events = self._parse_response(response, logics)
        
        # Set metadata
        for event in events:
            event.source = source or "unknown"
            event.content_hash = self._generate_content_hash(event)
        
        # Persist
        persisted = await self._save_events(events)
        
        return persisted
```

### 3. Batch Processing (`src/logic/batch_processor.py`)

```python
class EventBatchProcessor:
    """Process multiple news articles in batch"""
    
    def __init__(self):
        self.logic_service = LogicIdentificationService()
        self.event_service = EventExtractionService()
    
    async def process_batch(self, news_articles: List[Dict]) -> BatchResult:
        """
        Process batch of news articles
        
        Args:
            news_articles: List of {"text": str, "source": str, "date": date}
        
        Returns:
            BatchResult with logics and events extracted
        """
        all_logics = []
        all_events = []
        
        # Stage 1: Identify logics from all articles
        for article in news_articles:
            logics = await self.logic_service.identify_logics(
                article["text"],
                article.get("source")
            )
            all_logics.extend(logics)
        
        # Deduplicate logics
        unique_logics = self._deduplicate_logics(all_logics)
        logger.info(f"Identified {len(unique_logics)} unique logics")
        
        # Stage 2: Extract events using logic schema
        for article in news_articles:
            events = await self.event_service.extract_events(
                article["text"],
                unique_logics,
                article.get("source")
            )
            all_events.extend(events)
        
        logger.info(f"Extracted {len(all_events)} events")
        
        return BatchResult(logics=unique_logics, events=all_events)
```
